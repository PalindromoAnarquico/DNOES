    <script>
        const letreiro = document.querySelector('.letreiro');
        letreiro.style.overflow = 'hidden';
        letreiro.style.position = 'relative';
        letreiro.style.height = '80px';
        letreiro.s

        const imgs = document.querySelectorAll('.letreiro-img');
        imgs.forEach(img => {
            img.style.position = 'absolute';
            img.style.top = '0';
            img.style.height = '80px';
            img.style.whiteSpace = 'nowrap';
        });

        function deslizar() {
            const larguraTotal = Array.from(imgs).reduce((acc, img) => acc + img.width, 2);
            let pos = -larguraTotal;
            function frame() {
                pos += 2;
                let offset = pos;
                imgs.forEach(img => {
                    img.style.left = offset + 'px';
                    offset += img.width;
                });
                if (pos < letreiro.offsetWidth) {
                    requestAnimationFrame(frame);
                } else {
                    pos = -larguraTotal;
                    requestAnimationFrame(frame);
                }
            }
            frame();
        }
        // Espera todas as imagens carregarem
        let loaded = 0;
        imgs.forEach(img => {
            img.onload = () => {
                loaded++;
                if (loaded === imgs.length) deslizar();
            };
        });
    </script>
